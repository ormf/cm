[Function]
(dumposc oscfile . stream)

Prints the contents of oscfile to stream, which defaults to the standard output. Returns stream.

Examples:

Example 1. Printing the contents of an .osc file to the standard output. 

(defobject simple (scsynth)
  ((freq :initform 440)
   (dur :initform 1)
   (amp :initform .2)
   (pan :initform 0))
  (:parameters freq dur amp pan time))

(define (sc-simple-1 num)
  (process repeat num 
	   output (new simple :time (now) 
		       :freq (between 300 700)
		       :dur (between 10 20)
		       :amp .1
		       :pan (pickl '(-1.0 0 1.0)))
	   wait (between 2.0 3.0)))

(events (sc-simple-1 10) "sc-simple-1.osc" )
